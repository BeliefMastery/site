class DebugConsole {
  constructor() {
    this.enabled = localStorage.getItem('debug-mode') === 'true';
    this.logs = [];
    this.setupToggle();
  }
  
  setupToggle() {
    document.addEventListener('keydown', (e) => {
      if ((e.ctrlKey || e.metaKey) && e.key === 'd') {
        e.preventDefault();
        this.toggle();
      }
    });
  }
  
  toggle() {
    this.enabled = !this.enabled;
    localStorage.setItem('debug-mode', this.enabled);
    document.getElementById('debug-console').classList.toggle('visible');
  }
  
  log(category, message, data = null) {
    if (!this.enabled) return;
    
    const entry = {
      timestamp: new Date().toISOString(),
      category,
      message,
      data
    };
    
    this.logs.push(entry);
    this.render();
    
    // Also log to console for developer tools
    console.log(`[${category}]`, message, data);
  }
  
  render() {
    const console = document.getElementById('debug-log');
    console.innerHTML = this.logs.map(entry => `
      <div class="debug-entry">
        <span class="timestamp">${entry.timestamp}</span>
        <span class="category">${entry.category}</span>
        <span class="message">${entry.message}</span>
        ${entry.data ? `<pre>${JSON.stringify(entry.data, null, 2)}</pre>` : ''}
      </div>
    `).join('');
  }
  
  clear() {
    this.logs = [];
    this.render();
  }
}

/**
 * Unified Data Storage Manager
 * Handles localStorage with versioning, migration, and fallback
 */
class DataStore {
  constructor(namespace, version = '1.0.0') {
    this.namespace = namespace;
    this.version = version;
    this.storageAvailable = this.checkStorageAvailability();
  }
  
  checkStorageAvailability() {
    try {
      const test = '__storage_test__';
      localStorage.setItem(test, test);
      localStorage.removeItem(test);
      return true;
    } catch(e) {
      console.warn('localStorage not available, using fallback');
      return false;
    }
  }
  
  getKey(key) {
    return `${this.namespace}:${key}`;
  }
  
  save(key, data) {
    const payload = {
      version: this.version,
      timestamp: Date.now(),
      data: data
    };
    
    try {
      if (this.storageAvailable) {
        localStorage.setItem(this.getKey(key), JSON.stringify(payload));
        return true;
      } else {
        // Fallback to sessionStorage
        sessionStorage.setItem(this.getKey(key), JSON.stringify(payload));
        return true;
      }
    } catch (e) {
      console.error('Save failed:', e);
      return false;
    }
  }
  
  load(key) {
    try {
      const stored = this.storageAvailable 
        ? localStorage.getItem(this.getKey(key))
        : sessionStorage.getItem(this.getKey(key));
      
      if (!stored) return null;
      
      const payload = JSON.parse(stored);
      
      // Version checking
      if (payload.version !== this.version) {
        console.log(`Migrating data from ${payload.version} to ${this.version}`);
        return this.migrate(payload);
      }
      
      return payload.data;
    } catch (e) {
      console.error('Load failed:', e);
      return null;
    }
  }
  
  migrate(oldPayload) {
    // Override in subclasses for specific migration logic
    console.warn('No migration logic defined, returning old data');
    return oldPayload.data;
  }
  
  clear(key) {
    if (key) {
      localStorage.removeItem(this.getKey(key));
      sessionStorage.removeItem(this.getKey(key));
    } else {
      // Clear all keys for this namespace
      this.clearAll();
    }
  }
  
  clearAll() {
    const keysToRemove = [];
    
    for (let i = 0; i < localStorage.length; i++) {
      const key = localStorage.key(i);
      if (key.startsWith(`${this.namespace}:`)) {
        keysToRemove.push(key);
      }
    }
    
    keysToRemove.forEach(key => localStorage.removeItem(key));
  }
  
  export() {
    const data = {};
    
    for (let i = 0; i < localStorage.length; i++) {
      const key = localStorage.key(i);
      if (key.startsWith(`${this.namespace}:`)) {
        data[key] = localStorage.getItem(key);
      }
    }
    
    return {
      namespace: this.namespace,
      version: this.version,
      exportDate: new Date().toISOString(),
      data: data
    };
  }
  
  import(importData) {
    if (importData.namespace !== this.namespace) {
      console.error('Namespace mismatch');
      return false;
    }
    
    Object.entries(importData.data).forEach(([key, value]) => {
      localStorage.setItem(key, value);
    });
    
    return true;
  }
}

// Export for use in assessment engines

}

/**
 * Global Error Handler for Assessment Tools
 */
class ErrorHandler {
  constructor() {
    this.errors = [];
    this.userFacingErrors = [];
    this.setupGlobalHandlers();
  }
  
  setupGlobalHandlers() {
    window.addEventListener('error', (e) => {
      this.logError('RUNTIME_ERROR', e.message, {
        filename: e.filename,
        lineno: e.lineno,
        colno: e.colno
      });
    });
    
    window.addEventListener('unhandledrejection', (e) => {
      this.logError('PROMISE_REJECTION', e.reason, {
        promise: e.promise
      });
    });
  }
  
  logError(type, message, details = {}) {
    const error = {
      timestamp: new Date().toISOString(),
      type,
      message,
      details,
      userAgent: navigator.userAgent,
      url: window.location.href
    };
    
    this.errors.push(error);
    
    // Log to console in debug mode
    if (localStorage.getItem('debug-mode') === 'true') {
      console.error('[ERROR]', error);
    }
    
    // Send to debug console if available
    if (window.debug) {
      window.debug.log('ERROR', message, details);
    }
  }
  
  showUserError(message, duration = 5000) {
    const errorDiv = document.createElement('div');
    errorDiv.className = 'user-error';
    errorDiv.innerHTML = `
      <div class="error-content">
        <span class="error-icon">‚ö†Ô∏è</span>
        <span class="error-message">${message}</span>
        <button class="error-close" aria-label="Dismiss error">√ó</button>
      </div>
    `;
    
    document.body.appendChild(errorDiv);
    
    // Add close handler
    errorDiv.querySelector('.error-close').addEventListener('click', () => {
      errorDiv.remove();
    });
    
    // Auto-remove after duration
    setTimeout(() => {
      if (errorDiv.parentNode) {
        errorDiv.remove();
      }
    }, duration);
    
    this.userFacingErrors.push({ message, timestamp: Date.now() });
  }
  
  validate(condition, errorMessage) {
    if (!condition) {
      this.logError('VALIDATION_ERROR', errorMessage);
      this.showUserError(errorMessage);
      return false;
    }
    return true;
  }
  
  async tryAsync(fn, errorMessage = 'An error occurred') {
    try {
      return await fn();
    } catch (e) {
      this.logError('ASYNC_ERROR', e.message, { stack: e.stack });
      this.showUserError(errorMessage);
      return null;
    }
  }
  
  exportLog() {
    return {
      errors: this.errors,
      userFacingErrors: this.userFacingErrors,
      exportDate: new Date().toISOString()
    };
  }
}
