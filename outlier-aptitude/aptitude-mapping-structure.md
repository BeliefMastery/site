# Comprehensive Aptitude Mapping Structure

## Overview
This document defines how the 27 aptitude dimensions map to the 180 career paths, including personality archetype integration and market factor weighting.

## Aptitude Profile Templates by Career Category

### Template Structure
Each career receives a profile across all 27 dimensions:
- **Critical (0.7-1.0)**: Essential for success
- **Important (0.5-0.7)**: Significantly aids performance
- **Beneficial (0.3-0.5)**: Helpful but not required
- **Neutral (0.0-0.3)**: Minimal impact

## Integrated Scoring Formula

```javascript
function calculateCareerMatch(userProfile, career) {
  
  // 1. APTITUDE FIT (40% weight)
  let aptitudeScore = 0;
  let totalWeight = 0;
  
  for (const [dimension, requiredLevel] of Object.entries(career.aptitudeProfile)) {
    const userLevel = userProfile.aptitudes[dimension] || 0;
    const importance = requiredLevel; // Higher required = more important
    
    // Calculate match with non-linear scaling
    let match;
    if (userLevel >= requiredLevel) {
      match = 1.0; // Meets or exceeds requirement
    } else {
      // Penalty for shortfall, steeper for critical dimensions
      const shortfall = requiredLevel - userLevel;
      if (requiredLevel >= 0.7) { // Critical dimension
        match = Math.max(0, 1 - (shortfall * 2.0));
      } else if (requiredLevel >= 0.5) { // Important dimension
        match = Math.max(0, 1 - (shortfall * 1.5));
      } else { // Beneficial dimension
        match = Math.max(0, 1 - shortfall);
      }
    }
    
    aptitudeScore += match * importance;
    totalWeight += importance;
  }
  aptitudeScore = (aptitudeScore / totalWeight);
  
  // 2. ACUITY ALIGNMENT (15% weight)
  let acuityScore = 0;
  const userAcuity = [
    userProfile.acuity.primary,
    userProfile.acuity.secondary,
    userProfile.acuity.tertiary,
    ...userProfile.acuity.additional
  ];
  
  if (career.acuityFit.primary.includes(userProfile.acuity.primary)) {
    acuityScore += 1.0;
  } else if (career.acuityFit.secondary.includes(userProfile.acuity.primary)) {
    acuityScore += 0.7;
  } else if (career.acuityFit.viable.includes(userProfile.acuity.primary)) {
    acuityScore += 0.4;
  }
  
  // Bonus for secondary/tertiary alignment
  if (career.acuityFit.primary.includes(userProfile.acuity.secondary)) {
    acuityScore += 0.3;
  }
  if (career.acuityFit.secondary.includes(userProfile.acuity.tertiary)) {
    acuityScore += 0.2;
  }
  
  acuityScore = Math.min(1.0, acuityScore / 1.5); // Normalize
  
  // 3. PERSONALITY FIT (15% weight)
  let personalityScore = 0;
  let personalityMatches = 0;
  
  const personalityFactors = [
    'dominancePreference',
    'socialPreference',
    'riskTolerance',
    'stabilityPreference',
    'thinkingStyle',
    'workMode'
  ];
  
  for (const factor of personalityFactors) {
    const userPref = userProfile.personality[factor];
    const careerPref = career.personalityFit[factor];
    
    // Simple match scoring (can be refined with distance metrics)
    if (userPref === careerPref) {
      personalityMatches += 1.0;
    } else if (Math.abs(getLevelIndex(userPref) - getLevelIndex(careerPref)) <= 1) {
      personalityMatches += 0.6;
    } else {
      personalityMatches += 0.2;
    }
  }
  
  personalityScore = personalityMatches / personalityFactors.length;
  
  // 4. MARKET VIABILITY (20% weight)
  const growthScore = normalizeGrowth(career.marketData.growthRate5yr);
  const automationScore = career.marketData.automationResistance;
  const stabilityPenalty = career.marketData.skillEvolutionVelocity === 'very_high' ? 0.9 : 1.0;
  
  const marketScore = (
    (growthScore * 0.5) +
    (automationScore * 0.4) +
    (stabilityPenalty * 0.1)
  );
  
  // 5. ACCESSIBILITY (10% weight)
  let accessibilityScore = 1.0;
  
  // Education barrier
  const educationMatch = matchEducationLevel(
    userProfile.education,
    career.entryBarriers.educationMin
  );
  accessibilityScore *= educationMatch;
  
  // Experience barrier
  if (userProfile.yearsExperience < career.entryBarriers.experienceYears) {
    const experienceGap = career.entryBarriers.experienceYears - userProfile.yearsExperience;
    accessibilityScore *= Math.max(0.5, 1 - (experienceGap * 0.1));
  }
  
  // Capital barrier (for entrepreneurial paths)
  if (career.entryBarriers.capitalRequired > userProfile.availableCapital) {
    accessibilityScore *= 0.7;
  }
  
  // FINAL COMPOSITE SCORE
  const compositeScore = (
    aptitudeScore * 0.40 +
    acuityScore * 0.15 +
    personalityScore * 0.15 +
    marketScore * 0.20 +
    accessibilityScore * 0.10
  );
  
  // CONFIDENCE CALCULATION
  const confidence = calculateConfidence({
    profileCompleteness: userProfile.completeness,
    careerDataQuality: career.dataQuality,
    sampleSize: career.validationSampleSize
  });
  
  return {
    score: compositeScore,
    confidence: confidence,
    breakdown: {
      aptitude: aptitudeScore,
      acuity: acuityScore,
      personality: personalityScore,
      market: marketScore,
      accessibility: accessibilityScore
    },
    gaps: identifyGaps(userProfile, career),
    timeline: estimateTimeline(userProfile, career),
    nextSteps: generateNextSteps(userProfile, career)
  };
}

function normalizeGrowth(growthRate) {
  // Map growth rate to 0-1 score
  // 40%+ growth → 1.0
  // 15% growth → 0.8
  // 5% growth → 0.5
  // 0% growth → 0.3
  // -10% growth → 0.0
  
  if (growthRate >= 0.40) return 1.0;
  if (growthRate >= 0.15) return 0.6 + (growthRate - 0.15) / 0.25 * 0.4;
  if (growthRate >= 0.05) return 0.5 + (growthRate - 0.05) / 0.10 * 0.1;
  if (growthRate >= 0) return 0.3 + (growthRate / 0.05) * 0.2;
  return Math.max(0, 0.3 + (growthRate / 0.10) * 0.3);
}

function getLevelIndex(level) {
  const levels = {
    'low': 0, 'moderate': 1, 'moderate-high': 2, 'high': 3,
    'solitary': 0, 'independent': 1, 'collaborative': 2, 'highly-social': 3,
    'risk-averse': 0, 'calculated': 1, 'moderate': 2, 'high-risk': 3,
    'stability': 0, 'balanced': 1, 'variety': 2,
    'concrete': 0, 'practical': 1, 'conceptual': 2, 'abstract': 3,
    'competitive': 0, 'balanced': 1, 'collaborative': 2
  };
  return levels[level] || 1;
}

function matchEducationLevel(userEd, requiredEd) {
  const levels = {
    'none': 0,
    'high_school': 1,
    'certificate': 2,
    'associates': 3,
    'bachelors': 4,
    'masters': 5,
    'doctorate': 6,
    'phd': 7,
    'md': 7,
    'jd': 7
  };
  
  const userLevel = levels[userEd] || 0;
  const requiredLevel = levels[requiredEd] || 0;
  
  if (userLevel >= requiredLevel) return 1.0;
  
  const gap = requiredLevel - userLevel;
  return Math.max(0.3, 1 - (gap * 0.15));
}

function identifyGaps(userProfile, career) {
  const gaps = [];
  
  // Aptitude gaps
  for (const [dimension, required] of Object.entries(career.aptitudeProfile)) {
    const userLevel = userProfile.aptitudes[dimension] || 0;
    if (required >= 0.7 && userLevel < required) {
      gaps.push({
        type: 'critical_aptitude',
        dimension: dimension,
        current: userLevel,
        required: required,
        gap: required - userLevel,
        developable: isDevelopable(dimension)
      });
    }
  }
  
  // Education gap
  // Experience gap
  // Certification gaps
  
  return gaps;
}

function isDevelopable(dimension) {
  const highlyDevelopable = [
    'technical', 'organization', 'learning', 'self_regulation',
    'adaptability', 'metacognition', 'critical_thinking',
    'linguistic', 'clerical'
  ];
  
  const moderatelyDevelopable = [
    'systems', 'diagnostics', 'management', 'creativity',
    'scientific', 'eq', 'quantitative', 'achievement',
    'situational_judgment', 'mechanical'
  ];
  
  if (highlyDevelopable.includes(dimension)) return 'high';
  if (moderatelyDevelopable.includes(dimension)) return 'moderate';
  return 'low'; // Gf, Gc, spatial, processing, grit, resilience, artistic
}
```
